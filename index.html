    <script>

        window.onload = function() {

            const slides = document.querySelectorAll('.slide');

            const progressBar = document.getElementById('progressBar');

            const slideIndicator = document.getElementById('slideIndicator');

            let currentSlideIndex = 0;

            const slideDuration = 20000; // 20 seconds per slide

            let interval;

            let progressInterval;


            function showSlide(index) {

                // Hide all slides

                slides.forEach(slide => {

                    slide.classList.remove('active');

                    slide.style.opacity = '0'; // Ensure opacity reset

                });


                // Show the target slide

                slides[index].classList.add('active');

                slides[index].style.opacity = '1'; // Fade in


                // Update slide indicator

                slideIndicator.textContent = `${index + 1} / ${slides.length}`;

            }


            function updateProgressBar() {

                let width = 0;

                const increment = 100 / (slideDuration / 100); // Update every 100ms

                progressBar.style.width = '0%'; // Reset width for new slide


                clearInterval(progressInterval); // Clear previous progress interval


                progressInterval = setInterval(() => {

                    if (width >= 100) {

                        clearInterval(progressInterval);

                    } else {

                        width += increment;

                        progressBar.style.width = width + '%';

                    }

                }, 100); // Update every 100ms

            }


            function nextSlide() {

                currentSlideIndex = (currentSlideIndex + 1) % slides.length;

                showSlide(currentSlideIndex);

                updateProgressBar();

            }


            // Initial display and start

            showSlide(currentSlideIndex);

            updateProgressBar();

            interval = setInterval(nextSlide, slideDuration);

        };
